             #include <SoftwareSerial.h>
           
           // Connect the HM-10 TX to the Arduino RX on pin 2.
           // Connect the HM-10 RX to the Arduino TX on pin 12 through a voltage divider.
           SoftwareSerial BTserial(12, 3); // RX | TX
           
           //data type used to store a character value.
           char t; 
           
           //defines pins numbers for Motor A (Forward & Reverse)
           int pinA1 = 10;
           int pinA2 = 11;    //defines pins numbers for Motor B (Steering)
           int pinB1 = 6;
           int pinB2 = 5;
           
           void setup() {
           
             BTserial.begin(9600); //tells the Arduino to get ready to exchange messages with the Serial Monitor at a data rate of 9600 bits per second.
           
             Serial.begin(9600); //starts serial communication
             
             pinMode (pinA1, OUTPUT); //used to configure pinA1 to behave as an output.
             pinMode (pinA2, OUTPUT); //used to configure pinA1 to behave as an output.
           
             pinMode (pinB1, OUTPUT); //used to configure pinB1 to behave as an output.
             pinMode (pinB2, OUTPUT); //used to configure pinB2 to behave as an output.
           }
           
           void loop() {
             if (BTserial.available()) //gets the stored bytes from the serial port that are available for reading 
             {
               t = BTserial.read();
             }
             Serial.println(t); //print richting op seriële monitor
             if (t == 'F') {
               forward(200);
             }
             else if (t == 'B') {
               reverse(200);
             }
             else if (t == 'L') {
               analogWrite (pinB1, 600); //Writes an analog value (PWM wave) to a B1
               analogWrite(pinB2, LOW); //Writes an analog value (PWM wave) to a B2
               analogWrite(pinA1, LOW); //Writes an analog value (PWM wave) to a A1
               analogWrite(pinA2, LOW); //Writes an analog value (PWM wave) to a A2
             }
             else if (t == 'R') {
               analogWrite (pinB2, 600);  //Writes an analog value (PWM wave) to a B2
               analogWrite(pinB1, LOW);  //Writes an analog value (PWM wave) to a B1
               analogWrite(pinA1, LOW);  //Writes an analog value (PWM wave) to a A1
               analogWrite(pinA2, LOW);  //Writes an analog value (PWM wave) to a A2
             }
             else if (t == 'A') {
               stopcar();
             }
             else if (t =='D') {
               analogWrite(pinB1, LOW);  //Writes an analog value (PWM wave) to a B1
               analogWrite(pinB2, LOW);  //Writes an analog value (PWM wave) to a B2
               analogWrite(pinA1, LOW);  //Writes an analog value (PWM wave) to a A1
               analogWrite(pinA2, LOW);  //Writes an analog value (PWM wave) to a A2
             }
           
             delay(100); //wait 100ms
           }
           
           //car forward
           void forward(int s) {
             digitalWrite (pinA1, HIGH); //set pin A1 high
             digitalWrite (pinA2, LOW); //set pin A2 low
             analogWrite (pinA1, s);  //Writes an analog value (PWM wave) to a A1
           }
           
           
           void reverse(int b) {
             digitalWrite (pinA1, LOW); //set pin A1 low
             digitalWrite (pinA2, HIGH); //set pin A2 high
             analogWrite (pinA2, b);  //Writes an analog value (PWM wave) to a A2
           }
           
           void stopcar() {
             digitalWrite (pinA1, LOW); //set pin A1 low
             digitalWrite (pinA2, LOW); //set pin A1 low
           }
           
           
           void right() {
             analogWrite (pinB2, 600);  //Writes an analog value (PWM wave) to a B2
             delay(3000); //wait 3s
             digitalWrite (pinB2, LOW); //set pin B2 low
           }
           
           void left() {
             analogWrite (pinB1, 600);  //Writes an analog value (PWM wave) to a B1
             delay (5000); //wait 5s
             digitalWrite (pinB1, LOW); //set pin B1 low
           }
